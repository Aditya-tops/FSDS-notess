# Creating pie chart and customizing


import matplotlib.pyplot as plt

products = ["samsung", "iphone", "redmi"]
sales = [90, 40, 20]

c = ['y', 'r', 'b']

plt.pie(
	sales, 
	labels = products,
	colors = c,
	shadow = True,
	explode = (0.05, 0.05, 0.05),
	autopct = '%1.1f%%'
)

plt.axis('equal')
plt.legend()
plt.show()
