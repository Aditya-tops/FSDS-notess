# Steps from 1 to 7

print("Topic: XGBoost with Label encoding")
print()




print("Step 1: Importing the libraries")

from pandas import read_csv
from xgboost import XGBClassifier
from sklearn.preprocessing import LabelEncoder
from sklearn.model_selection import train_test_split






print("Step 2: Loading the dataset")

df = read_csv('iris.csv', header = None)
dataset = df.values






print("Step 3: Data preparation")

X = dataset[:, 0:4]
y = dataset[:, 4]


label_encoder = LabelEncoder()
label_encoder = label_encoder.fit(y)
label_encoded_y = label_encoder.transform(y)





print("Step 4: Splitting the dataset")

X_train, X_test, y_train, y_test = train_test_split(
    X,
    label_encoded_y, 
    test_size = 0.33, 
    random_state = 7
)




print("Step 5: Model creation")

model = XGBClassifier()






print("Step 6: Model training")

model.fit(X_train, y_train)



print("Step 7: Prediction")

print()
print(X_test[0:5])

predictions = model.predict(X_test)

print()
print(predictions[0:5])
